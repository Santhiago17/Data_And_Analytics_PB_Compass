CREATE EXTERNAL TABLE IF NOT EXISTS meubanco.exercicio_santhiago (
  nome STRING,
  sexo STRING,
  total INT,
  ano INT
)
ROW FORMAT SERDE 'org.apache.hadoop.hive.serde2.lazy.LazySimpleSerDe'
WITH SERDEPROPERTIES (
  'serialization.format' = ',',
  'field.delim' = ','
)
LOCATION 's3://exercicio-s3-santhiago/dados/';
